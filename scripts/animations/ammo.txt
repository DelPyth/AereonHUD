event HudLowAmmoPulse
{
  Animate AmmoInClip          FgColor "AmmoInClipLow"         Linear 0.0 0.01
  Animate AmmoInReserve       FgColor "AmmoInReserveLow"      Linear 0.0 0.01
  Animate AmmoNoClip          FgColor "AmmoNoClipLow"         Linear 0.0 0.01
  Animate AmmoInClipShadow    FgColor "AmmoInClipBgLow"       Linear 0.0 0.01
  Animate AmmoInReserveShadow FgColor "AmmoInReserveBgLow"    Linear 0.0 0.01
  Animate AmmoNoClipShadow    FgColor "AmmoNoClipBgLow"       Linear 0.0 0.01

  RunEvent HudLowAmmoPulseLoop 0.25
}

event HudLowAmmoPulseLoop
{
  RunEvent HudLowAmmoPulse 0.0
}

event HudLowAmmoPulseStop
{
  StopEvent HudLowAmmoPulse 0.0
  StopEvent HudLowAmmoPulseLoop 0.0

  Animate AmmoInClip          FgColor "AmmoInClip"         Linear 0.0 0.01
  Animate AmmoInReserve       FgColor "AmmoInReserve"      Linear 0.0 0.01
  Animate AmmoNoClip          FgColor "AmmoNoClip"         Linear 0.0 0.01
  Animate AmmoInClipShadow    FgColor "AmmoInClipBg"       Linear 0.0 0.01
  Animate AmmoInReserveShadow FgColor "AmmoInReserveBg"    Linear 0.0 0.01
  Animate AmmoNoClipShadow    FgColor "AmmoNoClipBg"       Linear 0.0 0.01
}
